Write a function that accepts parameter n and returns factorial of n.

        

            let factorial = (n) =>{
                if(n < 0){
                    console.log("The factorial of a negative nmber is not posibile.");
                }
                else if(n == 0){
                    console.log("This factorial of 0 is 1");
                }
                else{
                    let fact = 1
                    for(i=1; i <= n; i++){
                        fact = fact*i;
                    }
                    return fact;
                }
            }
            console.log(factorial(-1));
